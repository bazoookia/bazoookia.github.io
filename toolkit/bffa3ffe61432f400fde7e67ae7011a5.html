    shell常用命令

    童荣 4/14/2023

    介绍一些shell常用命令

    # curl

    • 发送二进制请求
    curl --data '@file_name'
    1
    • 获取执行信息
    curl -w '%{time_total}' -w '%{http_code}'
    1
    • 添加请求头
    curl -H 'Content-Type: application/json'
    1
    • 查看返回头
    curl -I
    1
    • dump header
    curl -D
    1
    • follow redirect
    curl -L
    1

    # git

    • 查询包含指定commit id的分支
    git branch --contains commitid
    1
    • clone大规模repo
    git clone --depth 1
git remote set-branches origin '*' or git remote set-branches origin branch_name # 关联远程分支
git fetch branch_name
git checkout branch_name
    1
    2
    3
    4
    • git查看指定commit id修改的文件
    git show commitid --name-only
git log -name-only
    1
    2
    • 配置全局ignore
    git config --global core.excludesfile ~/.gitignore
    1
    • 将文件标记为已解决
    git add -u
    1
    • merge回滚
    git revert current_commit -m [1|2] # 回滚一次merge
git merge --abort # 终止merge
    1
    2
    • http地址替换为ssh

    在~/.gitconfig中，增加如下配置

    [url "ssh://git@XXX/"]
    insteadOf = https://XXX/
    1
    2

    # gdb

    • 汇编代码
    set disassemble-next-line on  # set
disassemble					# 查看汇编代码
si							# step
i r $xx						# 查看寄存器中的值
    1
    2
    3
    4
    • 多线程
    set scheduler-locking step	# set
t xx							# 切换线程
info t						# 查看线程
t find xx						# 查找
    1
    2
    3
    4
    • 信号
    signal xx						# 向调试程序发送信号
    1
    • 记录调试日志
    set logging file xx
set logging on
    1
    2

    # llvm

    • 编译成可读指令文件(.ll)
    clang++ -std=c++20 -emit-llvm -g 1.cc -S -o 1.ll
    1
    • 编译成bitcode(.bc)
    clang++ -std=c++20 -emit-llvm -g 1.cc -o 1.bc
    1
    • .bc to .ll
    llvm-dis 1.bc
    1
    • .ll to .bc
    llvm-as 1.ll
    1
    • .bc to .s
    clang++ -S 1.bc -o 1.s
    1
